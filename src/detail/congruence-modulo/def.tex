\begin{frame}{同余}
	\label{cgrunt:def:cgrunt}

	\only<1->{若\(t\mid a-b\), 则称\(a\)与\(b\)模\(t\)同余, 记作\(a\equiv b\pmod t\)}

	\only<2->{反之则称\(a\)与\(b\)模\(t\)不同余, 记作\(a\nequiv b\pmod t\)}
\end{frame}


\begin{frame}{模 \(m\) 剩余类}
	\label{cgrunt:def:remainings}
	称

	\begin{equation}
		\Bbb{Z}_m:=\{\overline{0},\overline{1},\dots,\overline{m-1}\}
	\end{equation}

	为模 \(m\) 剩余类\pause

	其中 \(\overline{i}\in\{i+km\mid k\in\Bbb{Z}\}\)
\end{frame}
